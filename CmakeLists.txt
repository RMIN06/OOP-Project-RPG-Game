cmake_minimum_required(VERSION 3.10)
project(VroomsteinRPG LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Gather all source files in src/
file(GLOB_RECURSE SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp")
if(NOT SOURCES)
    message(FATAL_ERROR "No source files found in ${CMAKE_CURRENT_SOURCE_DIR}/src")
endif()

add_executable(${PROJECT_NAME} ${SOURCES})

target_include_directories(${PROJECT_NAME} PRIVATE "${CMAKE_CURRENT_SOURCE_DIR}/src")

# Don't use static SFML libraries
set(SFML_STATIC_LIBRARIES OFF)

find_package(SFML 3.1 COMPONENTS Graphics Window System Audio REQUIRED)

if(TARGET SFML::Graphics)
    target_link_libraries(${PROJECT_NAME} PRIVATE SFML::Graphics SFML::Window SFML::System SFML::Audio)
else()
    target_link_libraries(${PROJECT_NAME} PRIVATE sfml-graphics sfml-window sfml-system sfml-audio)
endif()

# Add Windows-specific libraries
if(WIN32)
    target_link_libraries(${PROJECT_NAME} PRIVATE winmm.lib)
endif()

# Post-build fix for m.lib on Windows
if(WIN32)
    add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
        COMMAND powershell.exe -Command "(Get-Content '${CMAKE_CURRENT_BINARY_DIR}/VroomsteinRPG.vcxproj') -replace 'm\\.lib;', '' | Set-Content '${CMAKE_CURRENT_BINARY_DIR}/VroomsteinRPG.vcxproj'"
    )
endif()

if(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/assets")
    add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_directory
        "${CMAKE_CURRENT_SOURCE_DIR}/assets"
        "$<TARGET_FILE_DIR:${PROJECT_NAME}>/assets"
    )
endif()
